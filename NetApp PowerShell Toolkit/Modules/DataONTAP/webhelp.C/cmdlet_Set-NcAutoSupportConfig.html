Cmdlet: Set-NcAutoSupportConfig


NAME
    Set-NcAutoSupportConfig
    
SYNOPSIS
    Modify AutoSupport configuration settings for the given node.
    
SYNTAX
    Set-NcAutoSupportConfig [-Node] <String> [-MailHosts <String[]>] [-From 
    <String>] [-To <String[]>] [-NoteTo <String[]>] [-PartnerAddress 
    <String[]>] [-Transport <String>] [-PostUrl <String>] [-PutUrl <String>] 
    [-ProxyUrl <String>] [-SupportAddress <String[]>] [-IsNodeInSubject 
    [<Boolean>]] [-IsNhtDataEnabled [<Boolean>]] [-IsPerfDataEnabled 
    [<Boolean>]] [-RetryInterval <Int32>] [-RetryCount <Int32>] 
    [-IsReminderEnabled [<Boolean>]] [-IsThrottleEnabled [<Boolean>]] 
    [-PayloadFormat <String>] [-PeriodicTxWindow <Int32>] [-MaxHttpSize 
    <String>] [-MaxSmtpSize <String>] [-IsPrivateDataRemoved [<Boolean>]] 
    [-IsLocalCollectionEnabled [<Boolean>]] [-IsEnabled [<Boolean>]] 
    [-IsSupportEnabled [<Boolean>]] [-IsOndemandEnabled [<Boolean>]] 
    [-IsOndemandRemoteDiagEnabled [<Boolean>]] [-OndemandServerUrl <String>] 
    [-OndemandPollingInterval <Int32>] [-MinimumPrivateDataLength <String>] 
    [-MaxDownloadRate <Int64>] [-MaxUploadRate <Int64>] 
    [-DisableOndemandDownload] [-ValidateDigitalCertificate [<Boolean>]] 
    [-Controller <NcController[]>] [-ZapiRetryCount <Int32>] 
    [<CommonParameters>]
    
    
DESCRIPTION
    Modify AutoSupport configuration settings for the given node.
